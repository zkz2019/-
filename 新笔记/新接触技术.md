- `style-resources-loader`

  - 可以用来配置全局 less 变量
    - `vue add style-resources-loader`
    - ```
        //vue.config.js
        const path = require('path')
        function resolve(dir) {
          return path.join(__dirname, dir)
        }
        const vueConfig = {
          ...
          pluginOptions: {
            'style-resources-loader': {
              preProcessor: 'less',
              patterns: [path.resolve(__dirname, 'src/assets/less/index.less')]
            }
          }
        }
        module.exports = vueConfig
      ```

- [loadsh](https://www.lodashjs.com/docs/lodash.add)

  - 工具函数库
    - `npm install lodash-webpack-plugin babel-plugin-lodash --save-dev`
    - ```
        //vue.config.js
        // lodash按需引入
        const LodashModuleReplacementPlugin = require('lodash-webpack-plugin')
        ...
        const vueConfig = {
          ...
          chainWebpack: (config) => {
            if (process.env.NODE.ENV === 'production') {
              config.plugin('loadshReplace').use(new        LodashModuleReplacementPlugin())
            }
          },
          ...
        }
          module.exports = vueConfig
      ```
    - 需要引入的地方直接引入使用: `import _ from "lodash";`

- [moment](http://momentjs.cn/docs/)

  - 时间处理库
    - `npm install moment`
    - 使用：`import moment from 'moment'`
    - `moment`体积较大，需要去除其他语言包 //优化
      ```
      //vue.config.js
        const vueConfig = {
        configureWebpack: {
          ...
          // webpack plugins
          plugins: [
            // Ignore all locale files of moment.js
            new webpack.IgnorePlugin(/^\.\/locale$/, /moment$/),
          ]
        },
        ...
      }
      ```
      然后在`main.js`里`import 'moment/locale/zh-cn'`
